Raw Data naming: ObsReward_B_[date separated by '_']_[time separated by '_'].csv

Persistent Block Info:
	- "BlockNum": "Started collecting. Block:{0}" or "Started pindropping. Block:{11}"
	- "BlockType": "Started {collecting}. Block:0" or "Started {pindropping}. Block:11"
	- "RoundNum": Zero Indexing - "Repositioned and ready to start block or round"
	- "CoinSetID": "coinsetID:{2} absolute and delta(local position)"

Meta Data: 
	- Coin Set ID's and respective Coin Point IDs: coinpoint{0}:  x: 12.27987 y: -1.0741 z: 7.540892 {deltax:-1.51 deltay:0 deltaz:2.71}
	- "CoinSet"
	- Pair ID # 
	- Pt ID #
	- A or B
	- Testing Date
	- Main or RR 

Block & Round Partitioning
	- Block Start: "Mark should happen if checked on terminal."
	- Block End: "finished current task"
	- Round Start (Collection or Pindropping) : "Repositioned and ready to start block or round"
	- TP1 Round End Num: "Finished pindrop round:1"

Spontaneous Block or Inter Block Events
	- 'Marks': "Sending Headset mark"
	- "BlockDisconnect": LogFile ends before a block ends "finished current task"

Chest Open Event: 
	Log File Data: 
		- "ChestOpened": N/A
		- "CoinPtIDCollected": "Other participant just collected coin: {1}"

Pin Drop Event Info:
	Log File Data
		- "PinDropped": "Other participant just dropped a new pin at -1.531 -1.557 -8.773"
		- "PinLoc": "Other participant just dropped a new pin at {-1.531 -1.557 -8.773}"
		- "ClosestLoc": "Dropped pin was dropped at 5.51 from chest 1 originally at {(-2.75, -1.56,	-3.42}):INCORRECT"

		- "ActualDist": "Dropped pin was dropped at {5.51} from chest 1 originally at (-2.75, -1.56,	-3.42):INCORRECT"
		- "GoodDrop": "Dropped pin was dropped at 5.51 from chest 1 originally at (-2.75, -1.56,	-3.42):{INCORRECT}" 
		- "CoinValueOrig": N/A
		- "POVote": "Observer chose CORRECT for this pindrop from the navigator" or "Observer chose INCORRECT for this pindrop from the navigator" or "Observer did not vote for this pindrop from the navigator"
		- "PO Vote Score": 
			- "POisCorrect":
				- If "GoodDrop" == "POVote"
			- "POisIncorrect"
				- If "GoodDrop" != "POVote"
			- "POdidntVote":
				- else
		- "CoinPointID": "Collected pin feedback coin: {1}"
		- "RPECoin": 
			- "PPE":     
				- If ("CoinSetID" == 2) AND ("CoinPointID" == 2)
			- "NPE":
				- If ("CoinSetID" == 3) AND ("CoinPointID" == 0)
			- "Normal": (all other coins not NPE or PPE)
				- If ("CoinSetID" == 1) 
				- OR ("CoinSetID" == 2) AND ("CoinPointID" == 0 OR "CoinPointID" == 1)
				- OR ("CoinSetID" == 3) AND ("CoinPointID" == 1 OR "CoinPointID" == 2)

		- "CoinValueCollected": "Collected feedback coin:{0.00} round reward: 0.00"

Parsing Variables:
- "PerfRound", "SwapVoteCorr", "RPECoin", "BlockDisconnect".


Summary Pin Drop Round Performance Info: 
	- "RoundReward": "Finished a perfect round with:{30.00} total reward: 0.00"
	- "TotalReward": "Finished a perfect round with:30.00 total reward: {0.00}"
	- "PerfRound": 
		- "Perfect": If "RoundReward" > 0
		- "Bad": If If "RoundReward" == 0
	- (TP2 Only) "SwapVote": "Observer says it was a {NEW} round." or "Observer says it was a {OLD} round."
	- (TP2 Only) "SwapVoteScore": 
		- "Correct": 
			- If ("CoinSetID" == 2 or "CoinSetID" == 3) AND "SwapVote" == "NEW"
			- OR("CoinSetID" == 1) AND "SwapVote" == "OLD"
		- "Incorrect":
			- If ("CoinSetID" == 1) AND "SwapVote" == "NEW"
			- OR ("CoinSetID" == 2 or "CoinSetID" == 3) AND "SwapVote" == "OLD"

XYZ Variables: 
	- ["HeadPosAnchored"	"EyeDirectionAnchored"	"FixationPointAnchored"]

Rotation Variables (Pitch Yaw Roll)
	- "HeadForthAnchored"

XY Variables: 
	- ["EyeLeft",	"EyeRight"]
