# === Histogram + KDE for All Numeric Columns ===

import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

df = pd.read_csv('summary_by_participant_role_coinset.csv')
numeric_columns = df.select_dtypes(include='number').columns

for column in numeric_columns:
    plt.figure()
    sns.histplot(df[column].dropna(), bins=20, kde=True)
    plt.title(f'Histogram of {column}')
    plt.xlabel(column)
    plt.ylabel('Frequency')
    plt.grid(True)
    plt.tight_layout()
    plt.show()

# === Split by Role/CoinSet with KDE Overlays ===

role_metrics = ['Number of Correct Swap Votes', '% Correct Swap Votes']
general_numeric_columns = [
    'Total Blocks Completed', 'Total Points Earned',
    'Total Session Time (sec)', 'Average Round Time (sec)',
    'Number of Total Swap Votes'
]

for column in general_numeric_columns:
    plt.figure()
    sns.histplot(data=df[df['currentRole'] == 'AN'], x=column, hue='coinSet', bins=20,
                 kde=True, multiple='layer', element='bars', alpha=0.6, palette='muted')
    plt.title(f'Histogram + KDE of {column} (AN only)')
    plt.tight_layout()
    plt.show()

for column in role_metrics:
    for group in ['currentRole', 'coinSet']:
        plt.figure()
        sns.histplot(data=df, x=column, hue=group, bins=20,
                     kde=True, multiple='layer', element='bars', alpha=0.6, palette='muted')
        plt.title(f'Histogram + KDE of {column} by {group}')
        plt.tight_layout()
        plt.show()

# === Violin Plots with Black Points ===

for metric in ['Average Round Time (sec)', '% Correct Swap Votes']:
    for group in ['coinSet', 'currentRole']:
        plt.figure()
        sns.violinplot(data=df, x=group, y=metric, palette='muted', inner=None, alpha=0.2)
        sns.stripplot(data=df, x=group, y=metric, color='black', jitter=0.25, size=3, alpha=0.6)
        plt.title(f'Violin Plot with Points: {metric} by {group}')
        plt.tight_layout()
        plt.show()

# === Scatterplot of Total Points vs Blocks (AN Only) ===

df_an = df[df['currentRole'] == 'AN']
plt.figure()
sns.scatterplot(data=df_an, x='Total Blocks Completed', y='Total Points Earned',
                hue='coinSet', style='coinSet', palette='Set2', s=60, edgecolor='black', alpha=0.8)
plt.title('Total Points vs. Total Blocks Completed (AN only)')
plt.tight_layout()
plt.show()

# === Compute and Plot % Correct PinDropVotes for POs ===

df['% Correct PinDropVotes'] = (df['Number of Correct PinDropVotes'] / df['Number of PinDropVotes']) * 100
df_po = df[df['currentRole'] == 'PO']

plt.figure()
sns.histplot(data=df_po, x='% Correct PinDropVotes', bins=20, kde=False, color='teal', alpha=0.7)
plt.title('% Correct PinDropVotes (PO only)')
plt.tight_layout()
plt.show()

plt.figure()
sns.violinplot(data=df_po, x='coinSet', y='% Correct PinDropVotes', palette='muted', inner=None, alpha=0.2)
sns.stripplot(data=df_po, x='coinSet', y='% Correct PinDropVotes', color='black', jitter=0.25, size=3, alpha=0.6)
plt.title('% Correct PinDropVotes by CoinSet (PO only)')
plt.tight_layout()
plt.show()
